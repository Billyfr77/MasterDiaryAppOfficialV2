/**
 * Formats currency values for display
 * @param {number} amount - The amount to format
 * @returns {string} Formatted currency string
 */
const formatCurrency = (amount) => {
  return new Intl.NumberFormat('en-US', {
    style: 'currency',
    currency: 'USD'
  }).format(amount || 0)
}

/**
 * Calculates item cost with overtime logic
 * @param {Object} item - The item to calculate cost for
 * @param {number} overtimeThreshold - Hours before overtime applies
 * @param {number} overtimeMultiplier - Overtime rate multiplier
 * @returns {number} Calculated cost
 */
const calculateItemCost = (item, overtimeThreshold, overtimeMultiplier) => {
  if (!item || item.type !== 'staff') {
    return (item?.rate || 0) * (item?.quantity || 0)
  }

  const regularHours = Math.min(item.quantity || 0, overtimeThreshold)
  const overtimeHours = Math.max(0, (item.quantity || 0) - overtimeThreshold)
  const regularCost = regularHours * (item.rate || 0)
  const overtimeCost = overtimeHours * (item.rate || 0) * overtimeMultiplier

  return regularCost + overtimeCost
}

/**
 * DraggableElement - Wrapper for drag-and-drop items
 * @param {Object} props - Component props
 * @param {Object} props.item - The item data to drag
 * @param {ReactNode} props.children - Child components
 */
const DraggableElement = ({ item, children }) => {
  const [{ isDragging }, drag] = useDrag(() => ({
    type: 'diary-item',
    item: item,
    collect: (monitor) => ({
      isDragging: monitor.isDragging(),
    }),
  }))

  const dragStyle = {
    opacity: isDragging ? 0.5 : 1,
    transform: isDragging ? 'scale(1.05) rotate(2deg)' : 'scale(1)',
    transition: 'all 0.3s cubic-bezier(0.4, 0, 0.2, 1)',
    cursor: isDragging ? 'grabbing' : 'grab',
    animation: isDragging ? 'none' : 'pulse 2s infinite'
  }

  return (
    <div ref={drag} style={dragStyle}>
      {children}
    </div>
  )
}

/**
 * DropZone - Area where items can be dropped
 * @param {Object} props - Component props
 * @param {string} props.entryId - ID of the entry this zone belongs to
 * @param {Function} props.onDrop - Callback when item is dropped
 * @param {ReactNode} props.children - Child components
 * @param {boolean} props.isHighlighted - Whether zone is highlighted
 */
const DropZone = ({ entryId, onDrop, children, isHighlighted }) => {
  const [{ isOver }, drop] = useDrop(() => ({
    accept: 'diary-item',
    drop: (item) => {
      onDrop(item, entryId)
      return undefined
    },
    collect: (monitor) => ({
      isOver: monitor.isOver(),
    }),
  }))

  const zoneStyle = {
    border: `3px ${isOver || isHighlighted ? 'solid' : 'dashed'} ${isOver || isHighlighted ? '#4ecdc4' : '#dee2e6'}`,
    borderRadius: '12px',
    background: isOver || isHighlighted
      ? 'linear-gradient(135deg, rgba(78, 205, 196, 0.2), rgba(68, 160, 141, 0.1))'
      : 'linear-gradient(135deg, rgba(248, 249, 250, 0.8), rgba(233, 236, 239, 0.4))',
    transition: 'all 0.4s cubic-bezier(0.4, 0, 0.2, 1)',
    cursor: 'pointer',
    minHeight: '80px',
    display: 'flex',
    alignItems: 'center',
    justifyContent: 'center',
    color: isOver || isHighlighted ? '#4ecdc4' : '#6c757d',
    padding: '24px',
    boxShadow: isOver || isHighlighted ? '0 8px 25px rgba(78, 205, 196, 0.3)' : '0 2px 10px rgba(0,0,0,0.05)',
    backdropFilter: 'blur(10px)'
  }

  return (
    <div ref={drop} style={zoneStyle}>
      {children || (
        <div style={{ textAlign: 'center', animation: 'bounce 2s infinite' }}>
          <Plus size={28} style={{ marginBottom: '8px' }} />
          <div style={{ fontSize: '16px', fontWeight: '500' }}>Drop items here to add to this entry</div>
          <div style={{ fontSize: '12px', opacity: 0.7, marginTop: '4px' }}>Unlimited entries supported</div>
        </div>
      )}
    </div>
  )
}

/**
 * DiaryEntry - Individual time entry with multimedia support
 * @param {Object} props - Component props
 * @param {Object} props.entry - Entry data
 * @param {Function} props.onUpdate - Update callback
 * @param {Function} props.onDelete - Delete callback
 * @param {Function} props.onDropItem - Drop item callback
 * @param {boolean} props.isDropTarget - Whether this is drop target
 * @param {Function} props.onAddPhotos - Add photos callback
 * @param {Function} props.onAddVoiceNote - Add voice note callback
 * @param {Function} props.onAddLocation - Add location callback
 * @param {Function} props.onPredictTime - AI time prediction callback
 * @param {string} props.theme - Current theme (dark/light)
 * @param {Function} props.formatCurrency - Currency formatter
 * @param {number} props.overtimeThreshold - Overtime threshold
 * @param {number} props.overtimeMultiplier - Overtime multiplier
 */
const DiaryEntry = ({
  entry, onUpdate, onDelete, onDropItem, isDropTarget,
  onAddPhotos, onAddVoiceNote, onAddLocation, onPredictTime, theme,
  formatCurrency, overtimeThreshold, overtimeMultiplier
}) => {
  const [isEditing, setIsEditing] = useState(false)
  const [noteText, setNoteText] = useState(entry.note || '')
  const [isRecording, setIsRecording] = useState(false)
  const [editingQuantityId, setEditingQuantityId] = useState(null)
  const [editingRateId, setEditingRateId] = useState(null)
  const [tempQuantity, setTempQuantity] = useState('')
  const [tempRate, setTempRate] = useState('')
  const mediaRecorderRef = useRef(null)
  const audioChunksRef = useRef([])
  const [showNotification, setShowNotification] = useState(false)
  const [notificationMessage, setNotificationMessage] = useState('')

  /**
   * Shows notification message
   * @param {string} message - Message to display
   */
  const showNotificationMessage = (message) => {
    setNotificationMessage(message)
    setShowNotification(true)
    setTimeout(() => setShowNotification(false), 3000)
  }

  /**
   * Saves note changes
   */
  const handleSaveNote = () => {
    onUpdate(entry.id, { note: noteText })
    setIsEditing(false)
    showNotificationMessage('Note saved')
  }

  /**
   * Saves quantity changes
   * @param {string} itemId - Item ID to update
   */
  const handleQuantitySave = (itemId) => {
    const newQuantity = parseFloat(tempQuantity) || 1
    const updatedItems = entry.items.map(item =>
      item.id === itemId ? { ...item, quantity: newQuantity } : item
    )
    onUpdate(entry.id, { items: updatedItems })
    setEditingQuantityId(null)
    setTempQuantity('')
    showNotificationMessage('Quantity updated')
  }

  /**
   * Saves rate changes
   * @param {string} itemId - Item ID to update
   */
  const handleRateSave = (itemId) => {
    const newRate = parseFloat(tempRate) || 0
    const updatedItems = entry.items.map(item =>
      item.id === itemId ? { ...item, rate: newRate } : item
    )
    onUpdate(entry.id, { items: updatedItems })
    setEditingRateId(null)
    setTempRate('')
    showNotificationMessage('Rate updated')
  }

  /**
   * Gets overtime breakdown for display
   * @param {Object} item - Item to analyze
   * @returns {Object|null} Overtime breakdown or null
   */
  const getOvertimeBreakdown = (item) => {
    if (item.type !== 'staff') return null
    const regularHours = Math.min(item.quantity || 0, overtimeThreshold)
    const overtimeHours = Math.max(0, (item.quantity || 0) - overtimeThreshold)
    return { regularHours, overtimeHours }
  }

  /**
   * Starts voice recording
   */
  const startRecording = async () => {
    try {
      const stream = await navigator.mediaDevices.getUserMedia({ audio: true })
      mediaRecorderRef.current = new MediaRecorder(stream)
      audioChunksRef.current = []

      mediaRecorderRef.current.ondataavailable = (event) => {
        audioChunksRef.current.push(event.data)
      }

      mediaRecorderRef.current.onstop = () => {
        const audioBlob = new Blob(audioChunksRef.current, { type: 'audio/wav' })
        const audioUrl = URL.createObjectURL(audioBlob)
        onAddVoiceNote(entry.id, audioUrl)
        stream.getTracks().forEach(track => track.stop())
        showNotificationMessage('Voice note added')
      }

      mediaRecorderRef.current.start()
      setIsRecording(true)
    } catch (err) {
      console.error('Error accessing microphone:', err)
      showNotificationMessage('Microphone access required')
    }
  }

  /**
   * Stops voice recording
   */
  const stopRecording = () => {
    if (mediaRecorderRef.current && isRecording) {
      mediaRecorderRef.current.stop()
      setIsRecording(false)
    }
  }

  /**
   * Gets current location
   */
  const getLocation = () => {
    if (navigator.geolocation) {
      navigator.geolocation.getCurrentPosition(
        (position) => {
          const location = {
            lat: position.coords.latitude,
            lng: position.coords.longitude,
            timestamp: new Date().toISOString()
          }
          onAddLocation(entry.id, location)
          showNotificationMessage('Location added')
        },
        (error) => {
          console.error('Error getting location:', error)
          showNotificationMessage('Location access required')
        }
      )
    } else {
      showNotificationMessage('Geolocation not supported')
    }
  }

  /**
   * Handles photo upload
   * @param {Event} e - File input event
   */
  const handlePhotoUpload = (e) => {
    const files = Array.from(e.target.files)
    files.forEach(file => {
      const reader = new FileReader()
      reader.onload = () => {
        onAddPhotos(e, entry.id)
        showNotificationMessage('Photo added')
      }
      reader.readAsDataURL(file)
    })
  }

  // Component styles
  const entryStyle = {
    background: theme === 'dark'
      ? 'linear-gradient(135deg, #2d3748 0%, #1a202c 100%)'
      : 'linear-gradient(135deg, #ffffff 0%, #fefefe 100%)',
    borderRadius: '16px',
    padding: '30px',
    border: `2px solid ${theme === 'dark' ? '#4a5568' : '#e9ecef'}`,
    boxShadow: theme === 'dark'
      ? '0 8px 32px rgba(0,0,0,0.3)'
      : '0 8px 32px rgba(0,0,0,0.1)',
    position: 'relative',
    overflow: 'hidden'
  }

  const contentStyle = {
    position: 'relative',
    zIndex: 1
  }

  return (
    <div style={entryStyle}>
      <style>{`
        .diary-entry::before {
          content: '';
          position: absolute;
          top: 0;
          left: 0;
          right: 0;
          height: 60px;
          background: linear-gradient(135deg, #4ecdc4 0%, #44a08d 100%);
          border-radius: 14px 14px 0 0;
        }
        .diary-content {
          position: relative;
          z-index: 1;
        }
        @media (max-width: 768px) {
          .diary-toolbar {
            width: 100% !important;
            margin-right: 0;
            margin-bottom: 20px;
          }
          .diary-main {
            flex-direction: column;
          }
          .summary-cards {
            grid-template-columns: 1fr !important;
          }
        }
      `}</style>

      <div className="diary-content">
        {/* Header Section */}
        <div style={{
          display: 'flex',
          justifyContent: 'space-between',
          alignItems: 'center',
          marginBottom: '20px'
        }}>
          <div style={{ display: 'flex', alignItems: 'center', gap: '12px' }}>
            <Calendar size={24} style={{ color: '#4ecdc4' }} />
            <h3 style={{
              margin: 0,
              color: theme === 'dark' ? '#e2e8f0' : '#495057',
              fontSize: '1.5rem',
              fontWeight: '600'
            }}>
              Entry #{entry.id}
            </h3>
            <span style={{
              background: entry.location ? '#28a745' : '#6c757d',
              color: 'white',
              padding: '4px 8px',
              borderRadius: '12px',
              fontSize: '0.75rem',
              fontWeight: '500'
            }}>
              {entry.time}
            </span>
          </div>

          <div style={{ display: 'flex', gap: '8px' }}>
            {entry.location && (
              <MapPin
                size={18}
                style={{ color: '#e74c3c' }}
                title={`Location: ${entry.location.lat.toFixed(4)}, ${entry.location.lng.toFixed(4)}`}
              />
            )}
            <button
              onClick={() => onDelete(entry.id)}
              style={{
                background: 'none',
                border: 'none',
                color: '#e74c3c',
                cursor: 'pointer',
                padding: '8px',
                borderRadius: '8px',
                transition: 'background 0.3s'
              }}
              onMouseEnter={(e) => e.target.style.background = '#ffeaea'}
              onMouseLeave={(e) => e.target.style.background = 'none'}
            >
              <Trash2 size={18} />
            </button>
          </div>
        </div>

        {/* Notes Section */}
        {isEditing ? (
          <div style={{ marginBottom: '20px' }}>
            <textarea
              value={noteText}
              onChange={(e) => setNoteText(e.target.value)}
              placeholder="Add your notes here..."
              style={{
                width: '100%',
                minHeight: '100px',
                padding: '12px',
                border: `1px solid ${theme === 'dark' ? '#4a5568' : '#ced4da'}`,
                borderRadius: '8px',
                background: theme === 'dark' ? '#2d3748' : '#f8f9fa',
                color: theme === 'dark' ? '#e2e8f0' : '#495057',
                fontSize: '14px',
                resize: 'vertical'
              }}
            />
            <div style={{ display: 'flex', gap: '8px', marginTop: '8px' }}>
              <button
                onClick={handleSaveNote}
                style={{
                  background: 'linear-gradient(135deg, #28a745 0%, #20c997 100%)',
                  color: 'white',
                  border: 'none',
                  padding: '8px 16px',
                  borderRadius: '8px',
                  cursor: 'pointer',
                  fontWeight: '500',
                  boxShadow: '0 4px 12px rgba(40, 167, 69, 0.3)'
                }}
              >
                Save Note
              </button>
              <button
                onClick={() => setIsEditing(false)}
                style={{
                  background: 'none',
                  border: `1px solid ${theme === 'dark' ? '#4a5568' : '#ced4da'}`,
                  color: theme === 'dark' ? '#e2e8f0' : '#495057',
                  padding: '8px 16px',
                  borderRadius: '8px',
                  cursor: 'pointer'
                }}
              >
                Cancel
              </button>
            </div>
          </div>
        ) : (
          <div style={{ marginBottom: '20px' }}>
            {entry.note ? (
              <p style={{
                margin: 0,
                color: theme === 'dark' ? '#a0aec0' : '#6c757d',
                lineHeight: '1.6',
                fontSize: '14px'
              }}>
                {entry.note}
              </p>
            ) : (
              <p style={{
                margin: 0,
                color: theme === 'dark' ? '#718096' : '#adb5bd',
                fontStyle: 'italic'
              }}>
                No notes added yet.
              </p>
            )}
            <button
              onClick={() => setIsEditing(true)}
              style={{
                marginTop: '8px',
                background: 'none',
                border: 'none',
                color: '#4ecdc4',
                cursor: 'pointer',
                fontSize: '14px',
                textDecoration: 'underline'
              }}
            >
              Edit Note
            </button>
          </div>
        )}

        {/* Multimedia Section */}
        <div style={{
          display: 'flex',
          flexWrap: 'wrap',
          gap: '16px',
          marginBottom: '24px',
          padding: '16px',
          background: theme === 'dark' ? 'rgba(45, 55, 72, 0.5)' : 'rgba(248, 249, 250, 0.8)',
          borderRadius: '8px',
          backdropFilter: 'blur(10px)'
        }}>
          <div style={{ display: 'flex', alignItems: 'center', gap: '8px' }}>
            <Camera size={16} style={{ color: '#4ecdc4' }} />
            <span style={{ fontSize: '12px', color: theme === 'dark' ? '#a0aec0' : '#6c757d' }}>
              Photos ({(entry.photos?.length || 0)})
            </span>
          </div>
          <div style={{ display: 'flex', alignItems: 'center', gap: '8px' }}>
            <Mic size={16} style={{ color: '#f39c12' }} />
            <span style={{ fontSize: '12px', color: theme === 'dark' ? '#a0aec0' : '#6c757d' }}>
              Voice ({(entry.voiceNotes?.length || 0)})
            </span>
          </div>
          <div style={{ display: 'flex', alignItems: 'center', gap: '8px' }}>
            <MapPin size={16} style={{ color: '#e74c3c' }} />
            <span style={{ fontSize: '12px', color: theme === 'dark' ? '#a0aec0' : '#6c757d' }}>
              Location ({entry.location ? 'Yes' : 'No'})
            </span>
          </div>
        </div>

        {/* Photos Display */}
        {entry.photos && entry.photos.length > 0 && (
          <div style={{ marginBottom: '24px' }}>
            <h4 style={{
              margin: '0 0 12px 0',
              color: theme === 'dark' ? '#e2e8f0' : '#495057',
              display: 'flex',
              alignItems: 'center',
              gap: '8px'
            }}>
              <Camera size={18} />
              Photos
            </h4>
            <div style={{
              display: 'grid',
              gridTemplateColumns: 'repeat(auto-fill, minmax(120px, 1fr))',
              gap: '12px'
            }}>

                  key={index}
                  src={photo}
                  alt={`Photo ${index + 1}`}
                  style={{
                    width: '100%',

                    objectFit: 'cover',

                    cursor: 'pointer',
                    transition: 'transform 0.3s',
                    border: `2px solid ${theme === 'dark' ? '#4a5568' : '#e9ecef'}`

            onClick={() => {
              document.getElementById(`photo-upload-${entry.id}`).click()
            }}
            style={{
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'center',
              gap: '8px',
              padding: '12px',
              background: 'linear-gradient(135deg, #667eea 0%, #764ba2 100%)',
              color: 'white',
              border: 'none',
              borderRadius: '8px',
              cursor: 'pointer',
              fontSize: '14px',
              fontWeight: '500',
              transition: 'all 0.3s ease',
              boxShadow: '0 4px 15px rgba(102, 126, 234, 0.4)'
            }}
            onMouseEnter={(e) => {
              e.target.style.transform = 'translateY(-2px)'
              e.target.style.boxShadow = '0 6px 20px rgba(102, 126, 234, 0.4)'
            }}
            onMouseLeave={(e) => {
              e.target.style.transform = 'translateY(0)'
              e.target.style.boxShadow = '0 4px 15px rgba(102, 126, 234, 0.4)'
            }}
          >
            <Upload size={16} />
            Add Photos
          </button>
          <input
            id={`photo-upload-${entry.id}`}
            type="file"
            multiple
            accept="image/*"
            onChange={handlePhotoUpload}
            style={{ display: 'none' }}
          />

          <button
            onClick={isRecording ? stopRecording : startRecording}

              display: 'flex',
              alignItems: 'center',
              justifyContent: 'center',
              gap: '8px',
              padding: '12px',

                ? 'linear-gradient(135deg, #e74c3c 0%, #c0392b 100%)'
                : 'linear-gradient(135deg, #f39c12 0%, #e67e22 100%)',
              color: 'white',
              border: 'none',
              borderRadius: '8px',
              cursor: 'pointer',
              fontSize: '14px',
              fontWeight: '500',
              transition: 'all 0.3s ease',
              transform: isRecording ? 'scale(1.05)' : 'scale(1)',
              boxShadow: isRecording
                ? '0 4px 15px rgba(231, 76, 60, 0.4)'
                : '0 4px 15px rgba(243, 156, 18, 0.4)'
            }}
          >
            {isRecording ? <MicOff size={16} /> : <Mic size={16} />}
            {isRecording ? 'Stop Recording' : 'Record Voice'}
          </button>

          <button
            onClick={getLocation}
            style={{
              display: 'flex',
              alignItems: 'center',
              justifyContent: 'center',

              fontSize: '14px',

                          onBlur={() => handleRateSave(item.id)}

                      padding: '8px',
                      background: theme === 'dark' ? 'rgba(243, 156, 18, 0.1)' : 'rgba(243, 156, 18, 0.05)',

                      border: '1px solid #f39c12'
                    }}>
                      <div style={{
                        fontSize: '11px',
                        color: '#f39c12',
                        marginBottom: '4px',
                        fontWeight: '600'
                      }}>
                        Overtime: {overtimeBreakdown.overtimeHours}hrs × {formatCurrency((item.rate || 0) * overtimeMultiplier)}
                      </div>
                      <div style={{
                        fontSize: '10px',
                        color: theme === 'dark' ? '#a0aec0' : '#6c757d'
                      }}>
                        Regular: {overtimeBreakdown.regularHours}hrs × {formatCurrency(item.rate || 0)}
                      </div>
                    </div>
                  )}

                  <div style={{
                    display: 'flex',
                    justifyContent: 'space-between',
                    alignItems: 'center',
                    paddingTop: '8px',
                    borderTop: `1px solid ${theme === 'dark' ? '#4a5568' : '#e9ecef'}`
                  }}>
                    <span style={{
                      fontSize: '12px',
                      color: theme === 'dark' ? '#a0aec0' : '#6c757d'
                    }}>
                      Total Cost
                    </span>
                    <span style={{
                      fontSize: '14px',
                      fontWeight: '600',
                      color: theme === 'dark' ? '#e2e8f0' : '#495057'
                    }}>
                      {formatCurrency(itemCost)}
                    </span>
                  </div>
                </div>
              )
            })}
          </div>
        </div>
      </div>

      {showNotification && (
        <div style={{
          position: 'absolute',
          top: '20px',
          right: '20px',
          background: '#28a745',
          color: 'white',
          padding: '8px 16px',
          borderRadius: '8px',
          boxShadow: '0 4px 12px rgba(40, 167, 69, 0.3)',
          zIndex: 1000,
          animation: 'slideInRight 0.3s ease-out'
        }}>
          {notificationMessage}
        </div>
      )}
    </div>
  )
}

  const fetchData = async () => {

              Overtime Threshold (hrs)

              }}
            />
          </div>

          <div>
            <label style={{
              display: 'block',
              color: theme === 'dark' ? '#e2e8f0' : '#495057',
              fontSize: '14px',
              fontWeight: '500',
              marginBottom: '8px'
            }}>
              Overtime Multiplier
            </label>
            <input
              type="number"
              step="0.1"
              value={overtimeMultiplier}
              onChange={(e) => onOvertimeMultiplierChange(parseFloat(e.target.value) || 1.5)}
              min="1"
              style={{
                width: '100%',
                padding: '8px 12px',
                border: `1px solid ${theme === 'dark' ? '#4a5568' : '#ced4da'}`,
                borderRadius: '6px',
                background: theme === 'dark' ? '#2d3748' : '#ffffff',
                color: theme === 'dark' ? '#e2e8f0' : '#495057',
                fontSize: '14px'
              }}
            />
          </div>
        </div>
      </div>
    </div>
  )
}

// ================================

// ================================

/**
 * PaintDiary - Main component for the revolutionary paint-your-day diary system
 * Features drag-and-drop time tracking, multimedia attachments, GPS, real-time calculations
 */
const PaintDiary = () => {
  // ================================
  // STATE MANAGEMENT
  // ================================
  const [selectedDate, setSelectedDate] = useState(new Date())
  const [diaryEntries, setDiaryEntries] = useState([])
  const [projects, setProjects] = useState([])
  const [selectedProject, setSelectedProject] = useState(null)
  const [totalCost, setTotalCost] = useState(0)
  const [totalRevenue, setTotalRevenue] = useState(0)
  const [isSaved, setIsSaved] = useState(true)
  const [isSaving, setIsSaving] = useState(false)
  const [dropTargetEntry, setDropTargetEntry] = useState(null)
  const [productivityScore, setProductivityScore] = useState(0)
  const [theme, setTheme] = useState('dark')
  const [margin, setMargin] = useState(20)
  const [overtimeThreshold, setOvertimeThreshold] = useState(8)
  const [overtimeMultiplier, setOvertimeMultiplier] = useState(1.5)
  const [currentDiaryId, setCurrentDiaryId] = useState(null)
  const [autoSaveInterval, setAutoSaveInterval] = useState(null)
  const [viewMode, setViewMode] = useState('daily')
  const [allDiaries, setAllDiaries] = useState([])

  // ================================
  // EFFECTS
  // ================================

  // Load existing diary when date changes
  useEffect(() => {
    loadExistingDiary(selectedDate.toISOString().split('T')[0])
  }, [selectedDate])

  // Load projects on mount
  useEffect(() => {
    api.get('/projects').then(res => setProjects(res.data.data || [])).catch(() => setProjects([]))
  }, [])

  // Recalculate totals when entries or margin change
  useEffect(() => {
    calculateTotals()
    calculateProductivityScore()
  }, [diaryEntries, margin])

  // Auto-save functionality
  useEffect(() => {
    if (!isSaved && diaryEntries.length > 0) {
      const interval = setInterval(() => {
        handleAutoSave()
      }, 30000) // Auto-save every 30 seconds
      setAutoSaveInterval(interval)
      return () => clearInterval(interval)
    } else if (autoSaveInterval) {
      clearInterval(autoSaveInterval)
      setAutoSaveInterval(null)
    }
  }, [isSaved, diaryEntries])

  // Keyboard shortcuts
  useEffect(() => {
    const handleKeyPress = (e) => {
      if (e.ctrlKey || e.metaKey) {
        switch (e.key) {
          case 's':
            e.preventDefault()
            handleSave()
            break
          case 'n':
            e.preventDefault()
            handleCreateEntry()
            break
          case 'e':
            e.preventDefault()
            handleExport()
            break
        }
      }
    }
    document.addEventListener('keydown', handleKeyPress)
    return () => document.removeEventListener('keydown', handleKeyPress)
  }, [])

  // ================================
  // BUSINESS LOGIC FUNCTIONS
  // ================================

  /**
   * Calculates totals for all diary entries
   */
  const calculateTotals = useCallback(() => {
    let cost = 0
    diaryEntries.forEach(entry => {
      entry.items.forEach(item => {
        cost += calculateItemCost(item, overtimeThreshold, overtimeMultiplier)
      })
    })
    const revenue = cost * (1 + margin / 100)
    setTotalCost(cost)
    setTotalRevenue(revenue)
  }, [diaryEntries, margin, overtimeThreshold, overtimeMultiplier])

  /**
   * Calculates productivity score based on efficiency
   */
  const calculateProductivityScore = useCallback(() => {
    if (diaryEntries.length === 0) {
      setProductivityScore(0)
      return
    }
    const totalHours = diaryEntries.reduce((sum, entry) =>
      sum + entry.items.reduce((itemSum, item) => itemSum + (item.duration || 1), 0), 0
    )
    const efficiency = totalRevenue > 0 ? (totalRevenue - totalCost) / totalRevenue : 0
    const score = Math.min(100, Math.max(0, (efficiency * 100) + (totalHours * 2)))
    setProductivityScore(Math.round(score))
  }, [diaryEntries, totalRevenue, totalCost])

  /**
   * Loads existing diary for selected date
   * @param {string} date - Date string in YYYY-MM-DD format
   */
  const loadExistingDiary = async (date) => {
    try {
      const response = await api.get(`/paint-diaries?date=${date}`)
      if (response.data && response.data.length > 0) {
        const diary = response.data[0]
        setCurrentDiaryId(diary.id)
        setDiaryEntries(diary.canvasData || [])
        setTotalCost(diary.totalCost || 0)
        setTotalRevenue(diary.totalRevenue || 0)
        setMargin(diary.margin || 20)
        setOvertimeThreshold(diary.overtimeThreshold || 8)
        setOvertimeMultiplier(diary.overtimeMultiplier || 1.5)
        setIsSaved(true)
      } else {
        setCurrentDiaryId(null)
        setDiaryEntries([])
        setTotalCost(0)
        setTotalRevenue(0)
        setIsSaved(true)
      }
    } catch (error) {
      console.error('Error loading diary:', error)
      setCurrentDiaryId(null)
      setDiaryEntries([])
      setTotalCost(0)
      setTotalRevenue(0)
      setIsSaved(true)
    }
  }

  // ================================
  // EVENT HANDLERS
  // ================================

  /**
   * Handles dropping an item onto a diary entry
   * @param {Object} item - Dropped item
   * @param {string} entryId - Target entry ID
   */
  const handleDropItem = useCallback((item, entryId) => {
    setDiaryEntries(prev => prev.map(entry =>
      entry.id === entryId ? {
        ...entry,
        items: [...(entry.items || []), {
          id: generateId(),
          type: item.type,
          name: item.name,
          rate: item.rate,
          quantity: 1,
          duration: item.type === 'staff' ? 1 : 0
        }]
      } : entry
    ))
    setDropTargetEntry(null)
    setIsSaved(false)
  }, [])

  /**
   * Updates a diary entry
   * @param {string} entryId - Entry ID to update
   * @param {Object} updates - Update object
   */
  const handleUpdateEntry = useCallback((entryId, updates) => {
    setDiaryEntries(prev => prev.map(entry =>
      entry.id === entryId ? { ...entry, ...updates } : entry
    ))
    setIsSaved(false)
  }, [])

  /**
   * Deletes a diary entry
   * @param {string} entryId - Entry ID to delete
   */
  const handleDeleteEntry = useCallback((entryId) => {
    setDiaryEntries(prev => prev.filter(entry => entry.id !== entryId))
    setIsSaved(false)
  }, [])

  /**
   * Adds photos to an entry
   * @param {Event} e - File input event
   * @param {string} entryId - Entry ID
   */
  const handleAddPhotos = useCallback((e, entryId) => {
    const files = Array.from(e.target.files)
    files.forEach(file => {
      const reader = new FileReader()
      reader.onload = () => {
        setDiaryEntries(entries => entries.map(entry =>
          entry.id === entryId ? { ...entry, photos: [...(entry.photos || []), reader.result] } : entry
        ))
        setIsSaved(false)
      }
      reader.readAsDataURL(file)
    })
  }, [])

  /**
   * Adds voice note to an entry
   * @param {string} entryId - Entry ID
   * @param {string} voiceNoteUrl - Voice note URL
   */
  const handleAddVoiceNote = useCallback((entryId, voiceNoteUrl) => {
    setDiaryEntries(entries => entries.map(entry =>
      entry.id === entryId ? { ...entry, voiceNotes: [...(entry.voiceNotes || []), voiceNoteUrl] } : entry
    ))
    setIsSaved(false)
  }, [])

  /**
   * Adds location to an entry
   * @param {string} entryId - Entry ID
   * @param {Object} location - Location object
   */
  const handleAddLocation = useCallback((entryId, location) => {
    setDiaryEntries(entries => entries.map(entry =>
      entry.id === entryId ? { ...entry, location } : entry
    ))
    setIsSaved(false)
  }, [])

  /**
   * AI-powered time prediction
   * @param {string} entryId - Entry ID
   */
  const handlePredictTime = useCallback((entryId) => {
    setDiaryEntries(entries => entries.map(entry => {
      if (entry.id !== entryId) return entry
      const staffCount = entry.items.filter(item => item.type === 'staff').length
      const equipmentCount = entry.items.filter(item => item.type === 'equipment').length
      const materialCount = entry.items.filter(item => item.type === 'material').length
      const baseDuration = staffCount * 2 + equipmentCount * 1.5 + materialCount * 0.5
      const predictedDuration = Math.max(0.25, baseDuration / Math.max(staffCount + equipmentCount + materialCount, 1))
      const updatedItems = entry.items.map(item => ({ ...item, duration: Math.max(0.25, predictedDuration) }))
      return { ...entry, items: updatedItems }
    }))
    setIsSaved(false)
  }, [])

  /**
   * Creates a new diary entry
   */
  const handleCreateEntry = useCallback(() => {
    const newEntry = {
      id: generateId(),
      time: new Date().toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' }),
      note: '',
      items: [],
      photos: [],
      voiceNotes: [],
      location: null
    }
    setDiaryEntries(prev => [newEntry, ...prev])
    setIsSaved(false)
  }, [])

  /**
   * Auto-saves the current diary
   */
  const handleAutoSave = useCallback(async () => {
    if (!isSaved && diaryEntries.length > 0) {
      try {
        const diaryData = {
          date: selectedDate.toISOString().split('T')[0],
          projectId: selectedProject?.id,
          canvasData: diaryEntries,
          totalCost,
          totalRevenue,
          margin,
          overtimeThreshold,
          overtimeMultiplier
        }
        if (currentDiaryId) {
          await api.put(`/paint-diaries/${currentDiaryId}`, diaryData)
        } else {
          const response = await api.post('/paint-diaries', diaryData)
          setCurrentDiaryId(response.data.id)
        }
        setIsSaved(true)
      } catch (error) {
        console.error('Auto-save error:', error)
      }
    }
  }, [isSaved, diaryEntries, selectedDate, selectedProject, totalCost, totalRevenue, margin, overtimeThreshold, overtimeMultiplier, currentDiaryId])

  /**
   * Saves the current diary manually
   */
  const handleSave = useCallback(async () => {
    setIsSaving(true)
    try {
      const diaryData = {
        date: selectedDate.toISOString().split('T')[0],
        projectId: selectedProject?.id,
        canvasData: diaryEntries,
        totalCost,
        totalRevenue,
        margin,
        overtimeThreshold,
        overtimeMultiplier
      }
      if (currentDiaryId) {
        await api.put(`/paint-diaries/${currentDiaryId}`, diaryData)
      } else {
        const response = await api.post('/paint-diaries', diaryData)
        setCurrentDiaryId(response.data.id)
      }
      setIsSaved(true)
      setIsSaving(false)
    } catch (error) {
      console.error('Save error:', error)
      setIsSaving(false)
    }
  }, [selectedDate, selectedProject, diaryEntries, totalCost, totalRevenue, margin, overtimeThreshold, overtimeMultiplier, currentDiaryId])

  /**
   * Exports diary data as JSON
   */
  const handleExport = useCallback(() => {
    const exportData = {
      date: selectedDate.toISOString().split('T')[0],
      project: selectedProject,
      entries: diaryEntries,
      totals: {
        cost: totalCost,
        revenue: totalRevenue,
        profit: totalRevenue - totalCost
      },
      settings: {
        margin,
        overtimeThreshold,
        overtimeMultiplier
      }
    }
    const blob = new Blob([JSON.stringify(exportData, null, 2)], { type: 'application/json' })
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url
    a.download = `diary-${selectedDate.toISOString().split('T')[0]}.json`
    a.click()
    URL.revokeObjectURL(url)
  }, [selectedDate, selectedProject, diaryEntries, totalCost, totalRevenue, margin, overtimeThreshold, overtimeMultiplier])

  /**
   * Exports diary data as CSV
   */
  const handleExportCSV = useCallback(() => {
    const csvData = [
      ['Entry ID', 'Time', 'Item Type', 'Item Name', 'Quantity', 'Rate', 'Cost', 'Revenue', 'Profit', 'Location', 'Photos', 'Voice Notes']
    ]
    diaryEntries.forEach(entry => {
      entry.items.forEach(item => {
        csvData.push([
          entry.id,
          entry.time,
          item.type,
          item.name,
          item.quantity,
          item.rate,
          calculateItemCost(item, overtimeThreshold, overtimeMultiplier),
          calculateItemCost(item, overtimeThreshold, overtimeMultiplier) * (1 + margin / 100),
          (calculateItemCost(item, overtimeThreshold, overtimeMultiplier) * (1 + margin / 100)) - calculateItemCost(item, overtimeThreshold, overtimeMultiplier),
          entry.location ? `${entry.location.lat.toFixed(4)}, ${entry.location.lng.toFixed(4)}` : ''
        ])
      })
    })
    const csvContent = csvData.map(row => row.map(cell => `"${cell}"`).join(',')).join('\n')
    const blob = new Blob([csvContent], { type: 'text/csv' })
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url
    a.download = `diary-${selectedDate.toISOString().split('T')[0]}.csv`
    a.click()
    URL.revokeObjectURL(url)
  }, [diaryEntries, selectedDate, margin, overtimeThreshold, overtimeMultiplier])

  /**
   * Toggles between light and dark theme
   */
  const toggleTheme = useCallback(() => {
    setTheme(prev => prev === 'light' ? 'dark' : 'light')
  }, [])

  /**
   * Fetches all diaries for overview
   */
  const fetchAllDiaries = useCallback(async () => {
    try {
      const response = await api.get('/paint-diaries')
      setAllDiaries(response.data || [])
    } catch (err) {
      console.error('Error fetching all diaries:', err)
    }
  }, [])

  /**
   * Deletes a diary entry
   * @param {string} diaryId - Diary ID to delete
   */
  const handleDeleteDiary = useCallback(async (diaryId) => {
    if (confirm('Are you sure you want to delete this diary?')) {
      try {
        await api.delete(`/paint-diaries/${diaryId}`)
        setAllDiaries(allDiaries.filter(d => d.id !== diaryId))
      } catch (err) {
        console.error('Error deleting diary:', err)
      }
    }
  }, [allDiaries])

  /**
   * Views a specific diary
   * @param {Object} diary - Diary object
   */
  const handleViewDiary = useCallback((diary) => {
    setSelectedDate(new Date(diary.date))
    setCurrentDiaryId(diary.id)
    setViewMode('daily')
  }, [])

  // ================================
  // RENDER LOGIC
  // ================================

  // Daily view mode
  if (viewMode === 'daily') {
    return (
      <DndProvider backend={HTML5Backend}>
        <div style={{
          minHeight: '100vh',
          background: theme === 'dark'
            ? 'linear-gradient(135deg, #0f1419 0%, #1a202c 100%)'
            : 'linear-gradient(135deg, #fefefe 0%, #f8f9fa 100%)',
          color: theme === 'dark' ? '#e2e8f0' : '#495057'
        }}>
          <style>{`
            .diary-page {
              background: ${theme === 'dark'
                ? 'linear-gradient(135deg, #2d3748 0%, #1a202c 100%)'
                : 'linear-gradient(135deg, #ffffff 0%, #fefefe 100%)'};
              border-radius: 16px;
              padding: 30px;
              border: 2px solid ${theme === 'dark' ? '#4a5568' : '#e9ecef'};
              box-shadow: ${theme === 'dark'
                ? '0 8px 32px rgba(0,0,0,0.3)'
                : '0 8px 32px rgba(0,0,0,0.1)'};
              position: relative;
              overflow: hidden;
            }
            .diary-content {
              position: relative;
              z-index: 1;
            }
            @media (max-width: 768px) {
              .diary-toolbar {
                width: 100% !important;
                margin-right: 0;
                margin-bottom: 20px;
              }
              .diary-main {
                flex-direction: column;
              }
              .summary-cards {
                grid-template-columns: 1fr !important;
              }
            }
          `}</style>

          <div style={{ maxWidth: '1400px', margin: '0 auto' }}>
            {/* Header Section */}
            <div style={{
              display: 'flex',
              justifyContent: 'space-between',
              alignItems: 'center',
              marginBottom: '32px'
            }}>
              <div>
                <h1 style={{
                  margin: '0 0 8px 0',
                  color: theme === 'dark' ? '#e2e8f0' : '#2c3e50',
                  fontSize: '3rem',
                  fontWeight: '700',
                  background: 'linear-gradient(135deg, #4ecdc4 0%, #667eea 100%)',
                  WebkitBackgroundClip: 'text',
                  WebkitTextFillColor: 'transparent',
                  backgroundClip: 'text'
                }}>
                  🎨 Paint Your Day Diary
                </h1>
                <p style={{
                  margin: 0,
                  color: theme === 'dark' ? '#a0aec0' : '#6c757d',
                  fontSize: '1.2rem'
                }}>
                  The ultimate construction time-tracking solution with AI, photos, voice, GPS & real-time cost calculations with overtime
                </p>
                <div style={{ display: 'flex', gap: '12px', marginTop: '16px' }}>
                  <button
                    onClick={() => setViewMode('daily')}
                    style={{
                      background: viewMode === 'daily' ? '#4ecdc4' : (theme === 'dark' ? '#2d3748' : '#f8f9fa'),
                      color: viewMode === 'daily' ? 'white' : (theme === 'dark' ? '#e2e8f0' : '#495057'),
                      border: 'none',
                      padding: '8px 16px',
                      borderRadius: '8px',
                      cursor: 'pointer',
                      fontWeight: '500'
                    }}
                  >
                    Daily View
                  </button>
                  <button
                    onClick={() => {
                      setViewMode('all')
                      fetchAllDiaries()
                    }}
                    style={{
                      background: viewMode === 'all' ? '#4ecdc4' : (theme === 'dark' ? '#2d3748' : '#f8f9fa'),
                      color: viewMode === 'all' ? 'white' : (theme === 'dark' ? '#e2e8f0' : '#495057'),
                      border: 'none',
                      padding: '8px 16px',
                      borderRadius: '8px',
                      cursor: 'pointer',
                      fontWeight: '500'
                    }}
                  >
                    All Diaries
                  </button>
                </div>
              </div>

              <div style={{ display: 'flex', gap: '12px', alignItems: 'center' }}>
                <DatePicker
                  selected={selectedDate}
                  onChange={(date) => {
                    setSelectedDate(date)
                    setCurrentDiaryId(null)
                  }}
                  dateFormat="EEEE, MMMM d, yyyy"
                  className="filter-input"
                  style={{
                    background: theme === 'dark' ? '#2d3748' : '#f8f9fa',
                    color: theme === 'dark' ? '#e2e8f0' : '#495057',
                    border: `1px solid ${theme === 'dark' ? '#4a5568' : '#ced4da'}`,
                    padding: '12px 16px',
                    borderRadius: '8px',
                    fontSize: '16px',
                    fontWeight: '500'
                  }}
                />

                <select
                  value={selectedProject?.id || ''}
                  onChange={(e) => setSelectedProject(projects.find(p => p.id === e.target.value) || null)}
                  style={{
                    background: theme === 'dark' ? '#2d3748' : '#f8f9fa',
                    color: theme === 'dark' ? '#e2e8f0' : '#495057',
                    border: `1px solid ${theme === 'dark' ? '#4a5568' : '#ced4da'}`,
                    padding: '12px 16px',
                    borderRadius: '8px',
                    fontSize: '16px',
                    fontWeight: '500'
                  }}
                >
                  <option value="">All Projects</option>
                  {projects.map(p => (
                    <option key={p.id} value={p.id}>{p.name}</option>
                  ))}
                </select>

                <DiaryToolbar
                  onExport={handleExport}
                  onExportCSV={handleExportCSV}
                  theme={theme}
                  onThemeToggle={toggleTheme}
                  margin={margin}
                  onMarginChange={setMargin}

                />

            </div>

            {/* Summary Cards */}
            <div style={{
              display: 'grid',
              gridTemplateColumns: 'repeat(auto-fit, minmax(220px, 1fr))',
              gap: '24px',
              marginBottom: '32px'
            }}>
              <div style={{
                background: 'linear-gradient(135deg, #e74c3c 0%, #c0392b 100%)',
                padding: '24px',
                borderRadius: '12px',
                color: 'white',
                textAlign: 'center',
                boxShadow: '0 8px 25px rgba(231, 76, 60, 0.3)',
                transition: 'all 0.3s ease'
              }}>
                <DollarSign size={28} style={{ marginBottom: '8px', opacity: 0.9 }} />
                <div style={{ fontSize: '2rem', fontWeight: 'bold' }}>{formatCurrency(totalCost)}</div>
                <div style={{ fontSize: '0.95rem', opacity: 0.9 }}>Total Cost</div>

              <div style={{
                background: 'linear-gradient(135deg, #4ecdc4 0%, #44a08d 100%)',
                padding: '24px',
                borderRadius: '12px',

                textAlign: 'center',
                boxShadow: '0 8px 25px rgba(78, 205, 196, 0.3)',
                transition: 'all 0.3s ease'

                background: 'linear-gradient(135deg, #f39c12 0%, #e67e22 100%)',
                padding: '24px',
                borderRadius: '12px',
                color: 'white',
                textAlign: 'center',
                boxShadow: '0 8px 25px rgba(243, 156, 18, 0.3)',
                transition: 'all 0.3s ease'
              }}>
                <BarChart3 size={28} style={{ marginBottom: '8px', opacity: 0.9 }} />
                <div style={{ fontSize: '2rem', fontWeight: 'bold' }}>{formatCurrency(totalRevenue - totalCost)}</div>
                <div style={{ fontSize: '0.95rem', opacity: 0.9 }}>Profit</div>

                padding: '24px',
                borderRadius: '12px',
                color: 'white',
                textAlign: 'center',
                boxShadow: '0 8px 25px rgba(155, 89, 182, 0.3)',
                transition: 'all 0.3s ease'
              }}>
                <Award size={28} style={{ marginBottom: '8px', opacity: 0.9 }} />
                <div style={{ fontSize: '2rem', fontWeight: 'bold' }}>{productivityScore}</div>
                <div style={{ fontSize: '0.95rem', opacity: 0.9 }}>Productivity Score</div>
              </div>
            </div>

            {/* Advanced Analytics */}
            {diaryEntries.length > 0 && (
              <div style={{
                background: theme === 'dark' ? '#1a202c' : 'white',
                padding: '24px',
                borderRadius: '12px',
                marginBottom: '32px',
                border: `1px solid ${theme === 'dark' ? '#4a5568' : '#e9ecef'}`,
                boxShadow: theme === 'dark' ? '0 4px 15px rgba(0,0,0,0.2)' : '0 4px 15px rgba(0,0,0,0.05)'
              }}>
                <h3 style={{
                  margin: '0 0 20px 0',
                  color: theme === 'dark' ? '#e2e8f0' : '#495057',
                  display: 'flex',
                  alignItems: 'center',
                  gap: '8px'
                }}>
                  <BarChart3 size={20} />
                  Advanced Analytics
                </h3>
                <div style={{ display: 'grid', gridTemplateColumns: 'repeat(auto-fit, minmax(250px, 1fr))', gap: '24px' }}>
                  <div>
                    <div style={{ display: 'flex', justifyContent: 'space-between', marginBottom: '6px' }}>
                      <span style={{ color: theme === 'dark' ? '#a0aec0' : '#6c757d' }}>Cost Efficiency</span>
                      <span style={{ fontWeight: '600', color: theme === 'dark' ? '#e2e8f0' : '#495057' }}>
                        {totalCost > 0 ? Math.round((totalRevenue / totalCost) * 100) : 0}%
                      </span>
                    </div>
                    <div style={{
                      height: '24px',
                      background: theme === 'dark' ? '#2d3748' : '#e9ecef',
                      borderRadius: '12px',
                      overflow: 'hidden'
                    }}>
                      <div style={{
                        height: '100%',
                        width: `${Math.min(100, (totalRevenue / Math.max(totalCost, 1)) * 100)}%`,
                        background: 'linear-gradient(90deg, #4ecdc4, #44a08d)',
                        transition: 'width 0.8s cubic-bezier(0.4, 0, 0.2, 1)',
                        borderRadius: '12px'
                      }}></div>
                    </div>
                  </div>
                  <div>
                    <div style={{ display: 'flex', justifyContent: 'space-between', marginBottom: '6px' }}>
                      <span style={{ color: theme === 'dark' ? '#a0aec0' : '#6c757d' }}>Profit Margin</span>
                      <span style={{ fontWeight: '600', color: theme === 'dark' ? '#e2e8f0' : '#495057' }}>
                        {margin}%
                      </span>
                    </div>
                    <div style={{
                      height: '24px',
                      background: theme === 'dark' ? '#2d3748' : '#e9ecef',
                      borderRadius: '12px',
                      overflow: 'hidden'
                    }}>
                      <div style={{
                        height: '100%',
                        width: `${margin}%`,
                        background: 'linear-gradient(90deg, #f39c12, #e67e22)',
                        transition: 'width 0.8s cubic-bezier(0.4, 0, 0.2, 1)',
                        borderRadius: '12px'
                      }}></div>
                    </div>
                  </div>
                  <div>
                    <div style={{ display: 'flex', justifyContent: 'space-between', marginBottom: '6px' }}>
                      <span style={{ color: theme === 'dark' ? '#a0aec0' : '#6c757d' }}>Entry Completion</span>
                      <span style={{ fontWeight: '600', color: theme === 'dark' ? '#e2e8f0' : '#495057' }}>
                        {Math.round(diaryEntries.reduce((sum, entry) => sum + entry.items.length, 0) / Math.max(diaryEntries.length, 1))}/entry avg
                      </span>
                    </div>
                    <div style={{
                      height: '24px',
                      background: theme === 'dark' ? '#2d3748' : '#e9ecef',
                      borderRadius: '12px',
                      overflow: 'hidden'
                    }}>
                      <div style={{
                        height: '100%',
                        width: `${Math.min(100, (diaryEntries.reduce((sum, entry) => sum + entry.items.length, 0) / Math.max(diaryEntries.length * 5, 1)) * 100)}%`,
                        background: 'linear-gradient(90deg, #667eea, #764ba2)',
                        transition: 'width 0.8s cubic-bezier(0.4, 0, 0.2, 1)',
                        borderRadius: '12px'
                      }}></div>
                    </div>
                  </div>
                </div>
              </div>
            )}

            {/* Main Content Area */}
            <div style={{
              display: 'grid',
              gridTemplateColumns: '280px 1fr',
              gap: '32px',
              alignItems: 'start'
            }}>
              <div className="diary-toolbar">
                <DiaryToolbar
                  onExport={handleExport}
                  onExportCSV={handleExportCSV}
                  theme={theme}
                  onThemeToggle={toggleTheme}
                  margin={margin}
                  onMarginChange={setMargin}
                  overtimeThreshold={overtimeThreshold}
                  onOvertimeThresholdChange={setOvertimeThreshold}
                  overtimeMultiplier={overtimeMultiplier}
                  onOvertimeMultiplierChange={setOvertimeMultiplier}
                  formatCurrency={formatCurrency}
                />
              </div>

              <div className="diary-main">
                <div style={{
                  display: 'flex',
                  justifyContent: 'space-between',
                  alignItems: 'center',
                  marginBottom: '24px'
                }}>
                  <div style={{ display: 'flex', alignItems: 'center', gap: '12px' }}>
                    <button
                      onClick={handleSave}
                      disabled={isSaved}
                      style={{
                        background: isSaved ? '#28a745' : 'linear-gradient(135deg, #667eea 0%, #764ba2 100%)',
                        color: 'white',
                        border: 'none',
                        padding: '12px 24px',
                        borderRadius: '8px',
                        cursor: isSaved ? 'not-allowed' : 'pointer',
                        fontWeight: '600',
                        boxShadow: isSaved ? '0 4px 12px rgba(40, 167, 69, 0.3)' : '0 4px 15px rgba(102, 126, 234, 0.4)',
                        transition: 'all 0.3s ease'
                      }}
                    >
                      {isSaving ? 'Saving...' : isSaved ? 'Saved' : 'Save Diary'}
                    </button>
                    {currentDiaryId && (
                      <span style={{
                        background: currentDiaryId ? '#d4edda' : '#fff3cd',
                        color: currentDiaryId ? '#155724' : '#856404',
                        padding: '8px 12px',
                        borderRadius: '6px',
                        fontSize: '14px',
                        fontWeight: '500'
                      }}>
                        {currentDiaryId ? `Loaded existing diary (ID: ${currentDiaryId})` : 'No saved diary for this date - create new entries'}
                      </span>
                    )}
                  </div>

                  <button
                    onClick={handleCreateEntry}
                    style={{
                      background: 'linear-gradient(135deg, #4ecdc4 0%, #44a08d 100%)',
                      color: 'white',
                      border: 'none',
                      padding: '12px 24px',
                      borderRadius: '8px',
                      cursor: 'pointer',
                      fontWeight: '600',
                      boxShadow: '0 6px 20px rgba(78, 205, 196, 0.4)',
                      transition: 'all 0.3s ease'
                    }}
                    onMouseEnter={(e) => {
                      e.target.style.transform = 'translateY(-2px)'
                      e.target.style.boxShadow = '0 8px 25px rgba(78, 205, 196, 0.4)'
                    }}
                    onMouseLeave={(e) => {
                      e.target.style.transform = 'translateY(0)'
                      e.target.style.boxShadow = '0 6px 20px rgba(78, 205, 196, 0.4)'
                    }}
                  >
                    <Plus size={18} style={{ marginRight: '8px' }} />
                    Add Entry
                  </button>
                </div>

                <div style={{
                  display: 'grid',
                  gridTemplateColumns: 'repeat(auto-fill, minmax(400px, 1fr))',
                  gap: '24px'
                }}>
                  {diaryEntries.length === 0 ? (
                    <div className="diary-page">
                      <div style={{
                        textAlign: 'center',
                        padding: '60px 20px',
                        color: theme === 'dark' ? '#a0aec0' : '#6c757d'
                      }}>
                        <Calendar size={48} style={{ marginBottom: '16px', opacity: 0.5 }} />
                        <h3 style={{ color: theme === 'dark' ? '#e2e8f0' : '#495057' }}>Your diary is empty</h3>
                        <p style={{ marginBottom: '24px', fontSize: '1.1rem' }}>
                          Drag items from the left to add them to your construction diary!
                        </p>
                        <button
                          onClick={handleCreateEntry}
                          style={{
                            background: 'linear-gradient(135deg, #4ecdc4 0%, #44a08d 100%)',
                            color: 'white',
                            border: 'none',
                            padding: '16px 32px',
                            borderRadius: '12px',
                            cursor: 'pointer',
                            fontSize: '16px',
                            fontWeight: '600',
                            boxShadow: '0 6px 20px rgba(78, 205, 196, 0.4)',
                            transition: 'all 0.3s ease'
                          }}
                          onMouseEnter={(e) => {
                            e.target.style.transform = 'translateY(-2px)'
                            e.target.style.boxShadow = '0 8px 25px rgba(78, 205, 196, 0.4)'
                          }}
                          onMouseLeave={(e) => {
                            e.target.style.transform = 'translateY(0)'
                            e.target.style.boxShadow = '0 6px 20px rgba(78, 205, 196, 0.4)'
                          }}
                        >
                          Create First Entry
                        </button>
                      </div>
                    </div>
                  ) : (
                    diaryEntries.map(entry => (
                      <DiaryEntry
                        key={entry.id}
                        entry={entry}
                        onUpdate={handleUpdateEntry}
                        onDelete={handleDeleteEntry}
                        onDropItem={handleDropItem}
                        isDropTarget={dropTargetEntry === entry.id}
                        onAddPhotos={handleAddPhotos}
                        onAddVoiceNote={handleAddVoiceNote}
                        onAddLocation={handleAddLocation}
                        onPredictTime={handlePredictTime}
                        theme={theme}
                        formatCurrency={formatCurrency}
                        overtimeThreshold={overtimeThreshold}
                        overtimeMultiplier={overtimeMultiplier}
                      />
                    ))
                  )}
                </div>
              </div>
            </div>
          </div>
        </div>
      </DndProvider>
    )
  }

  // All diaries view mode
  return (
    <div style={{
      minHeight: '100vh',
      background: theme === 'dark' ? '#1a1a2e' : '#f8f9fa',
      color: theme === 'dark' ? '#e2e8f0' : '#495057'
    }}>
      <div style={{ maxWidth: '1400px', margin: '0 auto', padding: '32px' }}>
        <h1>All Saved Diaries</h1>
        <button
          onClick={() => setViewMode('daily')}
          style={{
            marginBottom: '20px',
            padding: '10px 20px',
            background: '#4ecdc4',
            color: 'white',
            border: 'none',
            borderRadius: '8px'
          }}
        >
          Back to Daily View
        </button>
        {allDiaries.map(diary => (
          <div key={diary.id} style={{
            background: theme === 'dark' ? '#2d3748' : 'white',
            padding: '16px',
            marginBottom: '16px',
            borderRadius: '8px',
            border: `1px solid ${theme === 'dark' ? '#4a5568' : '#e9ecef'}`
          }}>
            <h3>{diary.date} - {diary.project?.name || 'No Project'}</h3>
            <p>Entries: {diary.canvasData?.length || 0}</p>
            <p>Total Cost: {formatCurrency(diary.totalCost || 0)}</p>
            <button
              onClick={() => handleViewDiary(diary)}
              style={{
                marginRight: '10px',
                padding: '8px 16px',
                background: '#667eea',
                color: 'white',
                border: 'none',
                borderRadius: '8px'
              }}
            >
              Edit
            </button>
            <button
              onClick={() => handleDeleteDiary(diary.id)}
              style={{
                padding: '8px 16px',
                background: '#e74c3c',
                color: 'white',
                border: 'none',
                borderRadius: '8px'
              }}
            >
              Delete
            </button>
          </div>
        ))}
      </div>
    </div>
  )
}
